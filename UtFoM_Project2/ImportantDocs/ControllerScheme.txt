Input.GetButtonDown("Dodge") = Dodge Button -> spacebar key, controller right trigger
Input.GetButtonDown("Confirm") = Confirm Button -> e key, NSPRO A button, PS circle button, XB B button
Input.GetButtonDown(light) = Light Button -> s key, NSPRO Y button, PS square button, XB X button
Input.GetButtonDown(launch) = Launch Button -> d key, NSPRO B button, PS X button, XB A button
Input.GetButtonDown(heavy) = Heavy Button -> f key, NSPRO X button, PS triangle button, XB Y button
Input.GetButtonDown("Heal") = Heal Button -> w key, controller right stick click
Input.GetButtonDown("Shortcut") = Shortcut Button -> a key, controller right bumper
Input.GetButtonDown("SwitchUp") = Switch Button + -> v key, controller dpad up, controller dpad right
Input.GetButtonDown("SwitchDown") = Switch Button - -> c key, controller dpad down, controller dpad left
Input.GetButtonDown("Run") = Run Button - ctrl key, controller left trigger
Input.GetButtonDown("Raft") = Raft Button - shift key, controller left bumper
Input.GetAxisRaw("Horizontal") = Move X -> left key + right key, left stick axis- + left stick axis+,
~controller dpad left + controller dpad right~
Input.GetAxisRaw("Vertical") = Move Y -> down key + up key, left stick axis- + left stick axis+, ~controller
dpad down + controller dpad up~
